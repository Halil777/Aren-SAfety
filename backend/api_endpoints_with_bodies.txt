Super Admin — Tenants
GET /api/super-admin/tenants
- Response: Tenant[]

GET /api/super-admin/tenants/:id
- Params: id
- Response: Tenant | null

POST /api/super-admin/tenants
- Body (JSON):
  {
    "slug": "lowercase-dash-slug",
    "title": "string",
    "adminLogin": "string (optional)",
    "adminPassword": "string(min 6, optional)",
    "adminEmail": "email (optional)",
    "adminFirstName": "string (optional)",
    "adminLastName": "string (optional)"
  }
- Response: Tenant

PATCH /api/super-admin/tenants/:id
- Params: id
- Body (JSON):
  {
    "slug": "lowercase-dash-slug (optional)",
    "title": "string (optional)"
  }
- Response: Tenant | null

- Params: id
- Response: {"deleted": number}

POST /api/super-admin/tenants/:id/admins
- Params: id
- Body (JSON):

    "login": "string",
    "password": "string(min 6)",
    "email": "email (optional)",
    "firstName": "string",
    "lastName": "string"
  }
- Response: Created admin object
